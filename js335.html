<script>
function clean(){
var canvas = document.getElementById('example');
var ctx = canvas.getContext('2d');
ctx.fillStyle = "white";
ctx.fillRect(0, 0, 640, 480);
ctx.closePath();
}
function er1(){
var canvas = document.getElementById('example');
var ctx = canvas.getContext('2d');
ctx.beginPath();
drawpoint();


ctx.moveTo(240, 0);
ctx.lineTo(240, 320);
ctx.moveTo(0, 160);
ctx.lineTo(480, 160);
ctx.arc(240, 160, 100, 0, 2 * Math.PI);
ctx.moveTo(450, 0);
ctx.lineTo(0, 450);
ctx.stroke();
x=Number(document.getElementById("x_znach").value);
y=Number(document.getElementById("y_znach").value);
if (x*x+y*y>5 && x>=5 && y<=-5 || x<=0 ) {
alert( 'Точка в области' );
} else {
alert( 'Точка не в области' );
}
}

function drawpoint() {
x=Number(document.getElementById("x_znach").value);
y=Number(document.getElementById("y_znach").value);
x=240+x*10;
y=160-y*10;
var canvas = document.getElementById('example');
var ctx = canvas.getContext('2d');
ctx.beginPath();
ctx.strokeStyle = 'green';
ctx.arc(x, y, 2, 0, 2 * Math.PI);
ctx.stroke();
}
